## **Layout Patterns**


### **Screen Structure (Standard Pattern)**


Every screen should follow this layout:


```dart
Scaffold(
  body: Container(
    decoration: backgroundGradient,
    child: Column(
      children: [
        // 1. Purple Gradient Header
        _buildHeader(),
        
        // 2. Overlapping White Card
        Expanded(
          child: SingleChildScrollView(
            child: Column(
              children: [
                Transform.translate(
                  offset: Offset(0, -32), // Overlap by 32px
                  child: _buildMainCard(),
                ),
                // Rest of content
                _buildContent(),
              ],
            ),
          ),
        ),
        
        // 3. Optional Footer
        _buildFooter(),
      ],
    ),
  ),
)
```


### **1. Header Pattern**


```dart
Widget _buildHeader() {
  return Container(
    decoration: BoxDecoration(
      gradient: LinearGradient(
        colors: [Color(0xFF9333EA), Color(0xFFA855F7)],
        begin: Alignment.centerLeft,
        end: Alignment.centerRight,
      ),
    ),
    padding: EdgeInsets.fromLTRB(24, 48, 24, 48),
    child: Stack(
      children: [
        // Decorative circles
        Positioned(
          top: -64,
          right: -64,
          child: Container(
            width: 128,
            height: 128,
            decoration: BoxDecoration(
              shape: BoxShape.circle,
              color: Colors.white.withOpacity(0.1),
            ),
          ),
        ),
        Positioned(
          bottom: -48,
          left: -48,
          child: Container(
            width: 96,
            height: 96,
            decoration: BoxDecoration(
              shape: BoxShape.circle,
              color: Colors.white.withOpacity(0.1),
            ),
          ),
        ),
        
        // Header content
        Column(
          children: [
            // Top row (back button, title, action)
            Row(
              mainAxisAlignment: MainAxisAlignment.spaceBetween,
              children: [
                _buildBackButton(),
                Text('Screen Title', style: headerWhite),
                _buildActionButton(),
              ],
            ),
            SizedBox(height: 24),
            
            // Icon circle
            Container(
              width: 80,
              height: 80,
              decoration: BoxDecoration(
                shape: BoxShape.circle,
                color: Colors.white,
                boxShadow: [
                  BoxShadow(
                    color: Colors.black.withOpacity(0.1),
                    blurRadius: 20,
                    offset: Offset(0, 10),
                  ),
                ],
              ),
              child: Icon(Icons.camera_alt, size: 40, color: primaryPurple),

    ),
  );
}
```

### **3. Priya Ma’am Card Pattern**


```dart
Widget _buildPriyaMaamCard(String message) {

      boxShadow: [
        BoxShadow(
          color: Color(0xFF9333EA).withOpacity(0.1),
          blurRadius: 12,
          offset: Offset(0, 4),
        ),
      ],
    ),
    padding: EdgeInsets.all(20),
    child: Row(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [

          height: 48,

            shape: BoxShape.circle,
```

## **Common Components**


### **Primary Button (Gradient)**


```dart
Widget _buildPrimaryButton({
  required String text,
  required VoidCallback onPressed,
  bool enabled = true,
}) {
  return Container(
    width: double.infinity,
    height: 56,
    decoration: BoxDecoration(
      gradient: enabled ? ctaGradient : null,
      color: enabled ? null : Color(0xFFE5E7EB),
      borderRadius: BorderRadius.circular(12),
      boxShadow: enabled ? [
        BoxShadow(
          color: Color(0xFF9333EA).withOpacity(0.3),
          blurRadius: 16,
          offset: Offset(0, 4),
        ),
      ] : [],
    ),
    child: Material(
      color: Colors.transparent,
      child: InkWell(
        onTap: enabled ? onPressed : null,
        borderRadius: BorderRadius.circular(12),
        child: Center(
          child: Text(
            text,
            style: TextStyle(
              fontSize: 16,
              fontWeight: FontWeight.w600,
              color: enabled ? Colors.white : Color(0xFF9CA3AF),
            ),
          ),
        ),
      ),
    ),
  );
}
```


### **Secondary Button (Outlined)**


```dart

        ),
      ),

```


### **Chip/Badge**


```dart
Widget _buildChip({
  required String text,
  required Color backgroundColor,
  required Color textColor,
  Color? borderColor,
}) {
  return Container(
    padding: EdgeInsets.symmetric(horizontal: 12, vertical: 6),
    decoration: BoxDecoration(
      color: backgroundColor,
      borderRadius: BorderRadius.circular(20),
      border: borderColor != null ? Border.all(color: borderColor) : null,
    ),
    child: Text(
      text,
      style: TextStyle(
        fontSize: 12,
        fontWeight: FontWeight.w600,
        color: textColor,
      ),
    ),
  );
}
```

## **Key Principles**


1. **Always use the purple gradient header** at the top
1. **Main content overlaps header by 32px** with white card
1. **Use Priya Ma’am card** for AI personality/messages
1. **Maintain consistent spacing** (24px screen padding, 16-20px card padding)
1. **Use gradient buttons** for primary actions
1. **Add subtle shadows** to create depth
1. **Round corners** everywhere (12-16px radius)
1. **Purple/pink brand colors** throughout
