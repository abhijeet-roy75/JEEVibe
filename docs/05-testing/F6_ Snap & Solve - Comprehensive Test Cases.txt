Example Performance Test (TC-PERF-004)
@Test
fun testOpenAIAPIResponseTime() {
    val startTime = System.currentTimeMillis()
    
    // Upload test image
    val testImage = loadTestResource("sample_question.jpg")
    snapSolveService.uploadAndSolve(testImage)
    
    // Wait for solution
    val solution = waitForSolution(timeout = 10000) // 10 seconds
    
    val endTime = System.currentTimeMillis()
    val duration = endTime - startTime
    
    // Assert under 10 seconds
    assertThat(duration).isLessThan(10000)
    assertThat(solution).isNotNull()
}
