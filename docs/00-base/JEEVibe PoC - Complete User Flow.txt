Snap Counter Logic (PoC)
Local Storage (No Auth):
javascript
// Store in localStorage (since no sign-in)
const snapData = {
  count: 2,
  limit: 5,
  lastResetDate: '2025-11-26', // Today's date
  history: [
    { timestamp: '2025-11-26T10:30:00', questionId: 'snap_1' },
    { timestamp: '2025-11-26T11:15:00', questionId: 'snap_2' }
  ]
};


// Check if needs reset
function getSnapsRemaining() {
  const today = new Date().toISOString().split('T')[0];
  const data = getLocalStorage('snapData');
  
  if (data.lastResetDate !== today) {
    // New day, reset counter
    return {
      count: 0,
      remaining: 5,
      resetToday: true
    };
  }
  
  return {
    count: data.count,
    remaining: 5 - data.count,
    resetToday: false
  };
}


// Increment counter
function useSnap() {
  const data = getLocalStorage('snapData');
  data.count += 1;
  data.history.push({
    timestamp: new Date().toISOString(),
    questionId: `snap_${Date.now()}`
  });
  setLocalStorage('snapData', data);
}
Reset Timer Display:
javascript
function getResetTime() {
  const now = new Date();
  const midnight = new Date();
  midnight.setHours(24, 0, 0, 0); // Next midnight
  
  const diff = midnight - now;
  const hours = Math.floor(diff / (1000 * 60 * 60));
  const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
  
  return `${hours}h ${minutes}m`;
}
```

## **Edge Cases to Handle:**


### **1. Hit 5-Snap Limit Mid-Session:**
```
User on Results Summary (used 4/5 snaps)
   ↓
Taps "Snap Another Question"
   ↓
Uses 5th snap (now 5/5)
   ↓
Completes practice questions
   ↓
On Results Summary, taps "Snap Another Question"
   ↓
Show "Daily Limit Reached" screen
2. User Closes App Mid-Flow:
javascript
// Save state to resume
function saveProgress(state) {
  localStorage.setItem('resumeState', JSON.stringify({
    screen: 'processing', // or 'practice', 'results'
    data: {...},
    timestamp: Date.now()
  }));
}


// On app launch, check for resume
function checkResume() {
  const state = localStorage.getItem('resumeState');
  if (state && (Date.now() - state.timestamp < 3600000)) { // 1 hour
    // Offer to resume
    showDialog('Resume where you left off?');
  }
}
